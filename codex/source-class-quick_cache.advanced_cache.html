282:             public function url_to_cache_path($url, $with_user_token = '', $with_version_salt = '', $flags = 0)
283:                 {
284:                     $cache_path        = ''; // Initialize.
285:                     $url               = trim((string)$url);
286:                     $with_user_token   = trim((string)$with_user_token);
287:                     $with_version_salt = trim((string)$with_version_salt);
288: 
289:                     if($url && strpos($url, '://') === FALSE)
290:                         $url = '//'.ltrim($url, '/');
291: 
292:                     if(!$url || !($url = parse_url($url)))
293:                         return ''; // Invalid URL.
294: 
295:                     if(!($flags & $this::CACHE_PATH_NO_SCHEME))
296:                         {
297:                             if(!empty($url['scheme']))
298:                                 $cache_path .= $url['scheme'].'/';
299:                             else $cache_path .= $this->is_ssl() ? 'https/' : 'http/';
300:                         }
301:                     if(!($flags & $this::CACHE_PATH_NO_HOST))
302:                         {
303:                             if(!empty($url['host']))
304:                                 $cache_path .= $url['host'].'/';
305:                             else $cache_path .= $_SERVER['HTTP_HOST'].'/';
306:                         }
307:                     if(!($flags & $this::CACHE_PATH_NO_PATH))
308:                         {
309:                             if(!empty($url['path']) && strlen($url['path'] = trim($url['path'], '\\/'." \t\n\r\0\x0B")))
310:                                 $cache_path .= $url['path'].'/';
311:                             else if(!($flags & $this::CACHE_PATH_NO_PATH_INDEX)) $cache_path .= 'index/';
312:                         }
313:                     $cache_path = str_replace('.', '-', $cache_path);
314: 
315:                     if(!($flags & $this::CACHE_PATH_NO_QUV))
316:                         {
317:                             if(!($flags & $this::CACHE_PATH_NO_QUERY))
318:                                 if(isset($url['query']) && $url['query'] !== '')
319:                                     $cache_path = rtrim($cache_path, '/').'.q/'.md5($url['query']).'/';
320: 
321:                             if(!($flags & $this::CACHE_PATH_NO_USER))
322:                                 if($with_user_token !== '') // Allow a `0` value if desirable.
323:                                     $cache_path = rtrim($cache_path, '/').'.u/'.str_replace(array('/', '\\'), '-', $with_user_token).'/';
324: 
325:                             if(!($flags & $this::CACHE_PATH_NO_VSALT))
326:                                 if($with_version_salt !== '') // Allow a `0` value if desirable.
327:                                     $cache_path = rtrim($cache_path, '/').'.v/'.str_replace(array('/', '\\'), '-', $with_version_salt).'/';
328:                         }
329:                     $cache_path = trim(preg_replace('/\/+/', '/', $cache_path), '/');
330:                     $cache_path = preg_replace('/[^a-z0-9\/.]/i', '-', $cache_path);
331: 
332:                     if(!($flags & $this::CACHE_PATH_NO_EXT))
333:                         $cache_path .= '.html';
334: 
335:                     return $cache_path;
336:                 }
