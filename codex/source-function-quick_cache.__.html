 224:             /**
 225:              * Converts a URL into a `cache/path`.
 226:              *
 227:              * @since 140422 First documented version.
 228:              *
 229:              * @param string  $url The input URL to convert.
 230:              * @param string  $with_user_token Optional user token (if applicable).
 231:              * @param string  $with_version_salt Optional version salt (if applicable).
 232:              * @param integer $flags Optional flags; a bitmask provided by `CACHE_PATH_*` constants.
 233:              *
 234:              * @return string The resulting `cache/path` based on the input `$url`.
 235:              */
 236:             public function build_cache_path($url, $with_user_token = '', $with_version_salt = '', $flags = 0)
 237:             {
 238:                 $cache_path        = ''; // Initialize.
 239:                 $url               = trim((string)$url);
 240:                 $with_user_token   = trim((string)$with_user_token);
 241:                 $with_version_salt = trim((string)$with_version_salt);
 242: 
 243:                 if($url && strpos($url, '://') === FALSE)
 244:                     $url = '//'.ltrim($url, '/');
 245: 
 246:                 if($url && strpos($url, '&amp;') !== FALSE)
 247:                     $url = str_replace('&amp;', '&', $url);
 248: 
 249:                 if(!$url || !($url = parse_url($url)))
 250:                     return ''; // Invalid URL.
 251: 
 252:                 if(!($flags & $this::CACHE_PATH_NO_SCHEME))
 253:                 {
 254:                     if(!empty($url['scheme']))
 255:                         $cache_path .= $url['scheme'].'/';
 256:                     else $cache_path .= $this->is_ssl() ? 'https/' : 'http/';
 257:                 }
 258:                 if(!($flags & $this::CACHE_PATH_NO_HOST))
 259:                 {
 260:                     if(!empty($url['host']))
 261:                         $cache_path .= $url['host'].'/';
 262:                     else $cache_path .= $_SERVER['HTTP_HOST'].'/';
 263:                 }
 264:                 if(!($flags & $this::CACHE_PATH_NO_PATH))
 265:                 {
 266:                     if(!empty($url['path']) && strlen($url['path'] = trim($url['path'], '\\/'." \t\n\r\0\x0B")))
 267:                         $cache_path .= $url['path'].'/';
 268:                     else if(!($flags & $this::CACHE_PATH_NO_PATH_INDEX)) $cache_path .= 'index/';
 269:                 }
 270:                 if($this->is_extension_loaded('mbstring') && mb_check_encoding($cache_path, 'UTF-8'))
 271:                     $cache_path = mb_strtolower($cache_path, 'UTF-8');
 272:                 $cache_path = str_replace('.', '-', strtolower($cache_path));
 273: 
 274:                 if(!($flags & $this::CACHE_PATH_NO_QUV))
 275:                 {
 276:                     if(!($flags & $this::CACHE_PATH_NO_QUERY))
 277:                         if(isset($url['query']) && $url['query'] !== '')
 278:                             $cache_path = rtrim($cache_path, '/').'.q/'.md5($url['query']).'/';
 279: 
 280:                     if(!($flags & $this::CACHE_PATH_NO_USER))
 281:                         if($with_user_token !== '') // Allow a `0` value if desirable.
 282:                             $cache_path = rtrim($cache_path, '/').'.u/'.str_replace(array('/', '\\'), '-', $with_user_token).'/';
 283: 
 284:                     if(!($flags & $this::CACHE_PATH_NO_VSALT))
 285:                         if($with_version_salt !== '') // Allow a `0` value if desirable.
 286:                             $cache_path = rtrim($cache_path, '/').'.v/'.str_replace(array('/', '\\'), '-', $with_version_salt).'/';
 287:                 }
 288:                 $cache_path = trim(preg_replace('/\/+/', '/', $cache_path), '/');
 289: 
 290:                 if($flags & $this::CACHE_PATH_ALLOW_WILDCARDS) // Allow `*`?
 291:                     $cache_path = preg_replace('/[^a-z0-9\/.*]/i', '-', $cache_path);
 292:                 else $cache_path = preg_replace('/[^a-z0-9\/.]/i', '-', $cache_path);
 293: 
 294:                 if(!($flags & $this::CACHE_PATH_NO_EXT))
 295:                     $cache_path .= '.html';
 296: 
 297:                 return $cache_path; // Do not filter.
 298:             }

 996:              *

1000:             {
1001:                 $dir_file = (string)$dir_file; // Force string value.
1002:                 if(!isset($dir_file[0])) return ''; // Catch empty string.
1003: 
1004:                 if(strpos($dir_file, '://' !== FALSE))  // A possible stream wrapper?

1043:             /**
1044:              * Deletes files from the cache directory (for the current host) that match a regex pattern.
1045:              *
1046:              * @since 140829 Implementing XML/RSS feed purging.
1047:              *
1048:              * @param string $regex A regex pattern; compares to each full file path.
1049:              *
1050:              * @return integer Total files deleted by this routine (if any).
1051:              *
1052:              * @throws \exception If unable to delete a file for any reason.
1053:              *
1054:              * @TODO @raamdev I think we could take the same concept introduced by this routine and use it to build others
1055:              *    that deal with purging/clearing/wiping; thereby centralizing this sort of job, making QC DRYer.
1056:              *    Also, this type of routine works to improve speed when running on a large MS network.
1057:              *
1058:              *    Suggested class members to come in a future release of QC.
1059:              *       - `purge_files_from_host_cache_dir()`
1060:              *       - `clear_files_from_host_cache_dir()`
1061:              *
1062:              *    Also, this class member (i.e. `delete_files_from_host_cache_dir()`) could be
1063:              *       used by many of the existing auto-purge routines. Thereby requiring less code
1064:              *       and speeding QC up overall; i.e. making it faster on large MS networks.
1065:              */
1066:             public function delete_files_from_host_cache_dir($regex)
1067:             {
1068:                 $counter = 0; // Initialize.
1069: 
1070:                 if(!($regex = (string)$regex))
1071:                     return $counter; // Nothing to do.
1072: 
1073:                 if(!is_dir($cache_dir = $this->cache_dir()))
1074:                     return $counter; // Nothing to do.
1075: 
1076:                 $host                 = $_SERVER['HTTP_HOST'];
1077:                 $host_base_dir_tokens = $this->host_base_dir_tokens();
1078:                 $cache_dir            = $this->n_dir_seps($cache_dir);
1079: 
1080:                 foreach(array('http', 'https') as $_host_scheme) // Consider all possible schemes.
1081:                 {
1082:                     $_host_url        = $_host_scheme.'://'.$host.$host_base_dir_tokens; // Base URL for this host (w/ MS support).
1083:                     $_host_cache_path = $this->build_cache_path($_host_url, '', '', $this::CACHE_PATH_NO_PATH_INDEX | $this::CACHE_PATH_NO_QUV | $this::CACHE_PATH_NO_EXT);
1084:                     $_host_cache_dir  = $this->n_dir_seps($cache_dir.'/'.$_host_cache_path);
1085: 
1086:                     /** @var $_dir_file \RecursiveDirectoryIterator For IDEs. */
1087:                     if($_host_cache_dir && is_dir($_host_cache_dir)) foreach($this->dir_regex_iteration($_host_cache_dir, $regex) as $_dir_file)
1088:                     {
1089:                         if(($_dir_file->isFile() || $_dir_file->isLink()) && ($_host_cache_dir !== $cache_dir || strpos($_dir_file->getSubpathname(), '/') !== FALSE))
1090:                             // Don't delete files in the immediate directory; e.g. `qc-advanced-cache` or `.htaccess`, etc.
